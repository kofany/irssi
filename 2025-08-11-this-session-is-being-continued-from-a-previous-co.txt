  static void sidebar_handle_mouse_click(int x, int y, int button_release)
  {
      (void)button_release;
      if (!sidebar_enabled || !sidebar_mouse_enable)
          return;
      /* Left column: window list */
      if (x >= 1 && x <= winlist_width) {
          int row, idx;
          GSList *swins, *t;

          /* coordinates are 1-based; our winlist window origin is at screen_reserved_top */
          row = y - screen_reserved_top; /* convert to 0-based row within winlist */
          if (row < 0) row = 0;
          swins = windows_get_sorted();
          idx = 1;
          for (t = swins; t != NULL; t = t->next, idx++) {
              if (idx == row + 1) { /* +1 because we display from row 1 */
                  WINDOW_REC *w = t->data;
                  window_set_active(w);
                  break;
              }
          }
          g_slist_free(swins);
          return;
      }

      /* Right column: nick list of active mainwindow */
      if (active_mainwin != NULL) {
          MAIN_WINDOW_REC *mw = active_mainwin;
          int nick_left = mw->last_column - nicklist_width + 1;
          int nick_right = mw->last_column;
          int nick_top = mw->first_line + mw->statusbar_lines_top;
          int nick_bottom = mw->last_line - mw->statusbar_lines_bottom;
          /* treat coordinates as 1-based absolute terminal coords */
          if (x >= nick_left && x <= nick_right && y >= nick_top && y <= nick_bottom) {
              int row;
              WINDOW_REC *w;
              row = y - nick_top; /* 0-based row within nicklist */
              w = mw->active;
              if (w != NULL && w->active != NULL) {
                  WI_ITEM_REC *item = w->active;
                  CHANNEL_REC *chan = CHANNEL(item);
                  if (chan != NULL) {
                      GSList *nicks, *t;
                      int idx;
                      nicks = nicklist_getnicks(chan);
                      nicks = g_slist_sort(nicks, nick_alpha_cmp);
                      t = nicks;
                      idx = 1;
                      for (; t != NULL; t = t->next, idx++) {
                          if (idx == row + 1) { /* +1 because we display from row 1 */
                              NICK_REC *nick = t->data;
                              if (nick != NULL && nick->nick != NULL) {
                                  /* open query with nick */
                                  signal_emit("command query", 1, nick->nick);
                              }
                              break;
                          }
                      }
                      g_slist_free(nicks);
                  }
              }
          }
      }
  }

  static void sig_gui_key_pressed_sidebar(gpointer keyp)
  {
      int key;
      if (!sidebar_mouse_enable) return;
      key = GPOINTER_TO_INT(keyp);
      if (mouse_parse_len == 0) {
          if (key == 0x1b) { /* ESC */
              mouse_parse_buf[mouse_parse_len++] = (char) key;
          }
          return;
      }
      if (mouse_parse_len == 1) {
          if (key == '[') {
              mouse_parse_buf[mouse_parse_len++] = (char) key;
          } else {
              mouse_parse_len = 0;
          }
          return;
      }
      if (mouse_parse_len == 2) {
          if (key == '<') {
              mouse_parse_buf[mouse_parse_len++] = (char) key;
          } else {
              mouse_parse_len = 0;
          }
          return;
      }
      /* accumulate until M/m or buffer full */
      if (mouse_parse_len < (int) sizeof(mouse_parse_buf) - 1) {
          mouse_parse_buf[mouse_parse_len++] = (char) key;
          mouse_parse_buf[mouse_parse_len] = '\0';
          if (key == 'M' || key == 'm') {
              int b, x, y;
              if (parse_mouse_triplet(mouse_parse_buf, &b, &x, &y)) {
                  /* act on left button press/release */
                  int is_release = (key == 'm');
                  int btn = b & 3; /* 0 left, 1 middle, 2 right typically */
                  if (btn == 0) sidebar_handle_mouse_click(x, y, is_release);
              }
              mouse_parse_len = 0;
          }
      } else {
          mouse_parse_len = 0;
      }
  }
